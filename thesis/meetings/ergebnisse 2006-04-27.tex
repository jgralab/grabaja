\begin{itemize}
	\item{Es soll ein Metamodell für die feingranulare Abbildung von Java-Quelltexten erstellt werden:}
	\begin{itemize}
		\item{Übernehmen von fremden Modellen nur insoweit, wie die Quelle den eigenen Aufwand mindert.}
		\item{Übernehmen von fremden Modellen nur insoweit, wie die Quelle unserer eigenen Struktur entspricht oder ohne größeren Aufwand daran angepasst werden kann.}
	\end{itemize}
	\item{Überarbeitung des Dokumentes antlr.tex:}
	\begin{itemize}
		\item{Eigener Darstellungsstil für Grammatiken, welcher sich von Codedarstellungen unterscheidet.}
		\item{Anderer Font anstelle von Hervorhebung im gleichen Font.}
		\item{Kleinere Schriftgröße bei der Zeilennummerierung der Listings.}
		\item{Das verwendete Beispeil zu Anfang einführen und die Beispiele durchgängig darauf beziehen.}
		\item{Abschnitt 2.6: Das Beispiel aus 2.5 verwenden und Input/Output auflisten.}
		\item{Abschnitt 3.1: Alternative für Erzeugung des Offset (Sammeln der Informationen in einem zweiten Durchlauf) nennen.}
		\item{Abschnitt Speicherung des Offset: Die beiden Alternativen vorher nennen und Vorgehen aufzeigen.}
		\item{Abschnitt 3: Klassendiagramm und Aktivitätsdiagramm von einander trennen.}
	\end{itemize}
